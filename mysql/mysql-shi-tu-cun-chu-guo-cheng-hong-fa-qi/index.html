视图

介绍

视图(view)是一张虚拟表。试图表中的数据并不在数据库中真实的存在，行和列数据来自定义视图的查询中使用的表，并且是在使用视图时动态生成。

视图只保存了查询的SQL逻辑，不保存查询结果。

在创建视图时，主要的工作就是这条SQL的查询语句上。

作用

简单

简化用户对数据的操作，简化他们的操作。

安全

数据库授权，最细力度只能到表，无法授权到固定的行和列上。通过视图授权用户只能查询和修改他们见到的字段。

数据独立

帮助用户屏蔽基础表带来的变化。

创建视图

create [or Replace] view 视图名 as 
select语句 where 条件
with [local | cascaded] check option;

#with [cascaded|local] check option
#检查选项

视图检查选项

使用with [local | cascaded] check option 子句创建视图时，Mysql会通过视图检查正在更改的每行，列入：插入，更新，删除，以使其符合视图的定义。Mysql允许基于一个视图创建另一个试图，还会检查视图中的规则保持一致性。为了确定检查范围mysql提供了两个选项，CASCADED和LOCAL，默认值是CASCADED。

cascaded
with [local | cascaded] check option;
#with [cascaded|local] check option
#检查选项加上后在操作数据时会检查当前视图条件还会检查所依赖的子视图条件。

create view v1 as select * from table where id <= 20 

create view v2 as select * from v1 where id >= 10 cascaded check option;
#cascaded 级联.v2视图的条件在v1视图同样生效
insert into v1 values(5,'aa')
#无法插入v1，因为v2的条件是>=
local

操作数据时，自己的条件和所依赖的子视图条件都会检查，只判断校验自己的条件，但是如果子视图也定了检查选项，那么也会效验和判断子视图条件。

create view v1 as select * from table where id <= 20 

create view v2 as select * from v1 where id >= 10 local check option;

insert into v2 values(2,15)
#可以插入
insert into v2 values(2,25)
#可以插入
insert into v1 values(2,5)
#无法插入

删除视图

drop view [if exists]视图名 [,视图名];

查看视图

select * from 视图名;
#查看视图数据

show create view 视图名称;
#查看创建视图语句

更新视图

要使视图可更新，视图中的行和基础表中的行必须存在一对一的关系。如果试图包含以下任意一行，则该视图不可更新。

1.聚合函数

2.DISTINCT

3.GROUP BY

4.HAVING

5.UNION或者UNION ALL

create or replace view 视图名称 
as select语句
#方式一

alter view 视图名 (字段名) as select语句
#方式二

存储过程

介绍

存储过程是事先经过编译并存储在数据库中的一段SQL语句集合。减少数据在数据库和服务器之间的传输，提高数据处理的效率是有好处的。

存储过程思想上很简单，就是数据库SQL语言层面的代码封装与重用。

特点

封装和复用

可以接受参数，也可以返回数据

减少网络交互，效率提升

基本语法

创建存储过程

create procedure 存储过程名称([参数列表])
Begin
SQL语句
END;

调用存储过程

CALL 存储过程名称
call 过程名[参数]

查看存储过程

查询指定数据库的存储过程及状态信息

select * from information_schema.ROUTINES where routine_schema = '数据库名称';
#查询指定数据库的存储过程及状态信息

查询某个存储过程的定义

show create procedure 存储过程名称

删除存储过程

DROP procedure [if exists] 存储过程名称;

设置分隔符

delimiter $$

# '$$' 替换原来的 ';'
delimiter ;

变量

系统变量

系统变量是Mysql服务器提供的，属于服务器层面。分为全局变量（GLOBAL），会话变量（SESSION）。

默认会话级别。

查看系统变量
show [session|GLOBAL] variables;
#查看所有系统变量

show [session|GLOBAL] variables like '....'
#like方式模糊匹配查找系统变量

select @@[SESSION|GLOBAL] 系统变量名;
#查看指定变量值
设置系统变量
set [SESSION|GLOBAL] 系统变量名=值;
set @@[Session|GLOBAL] 系统变量名=值;
#全局变量参数设置后，Mysql重启后会失效

用户变量

mysql用户变量，mysql中用户变量不用提前申明，在用的时候直接用“@变量名”使用就可以了。作用域为当前连接。

声明用户变量
set @nvar_name= expr[,@var_name=expr]...;
set @var_name := expr[,@var_name=expr]...;
#推荐使用第二种，因为只有=没有==,方便区分

select @var_name := expr;
#也可以根据select来定义用户变量
select 字段名 into @var_name from 表名;
#将sql查询出的字段名称赋值给 @var_name;
查看用户变量
select @var_name [,var_name,...]
select @没赋值的变量
#默认是NULL

局部变量

是根据需要定义的在局部生效的变量。用DECLARE声明。

可用作存储过程中的局部变量和输入参数。

局部变量只能用在begin/end语句块中

声明局部变量
DECLARE 变量名 变量类型 [DEFAULT...];
declare 变量名 varchar(20) [default 0];
赋值局部变量
set 变量名=值;
set 变量名:=值;
seelct 字段名 into 变量名 from 表名 where 条件 ;
查看局部变量
select var_name;
#不用@

elseif	条件2	then	结果2
elseif	条件3	then	结果3

case流程控制

语法一

case case_value
    when when_value1 then statement_list1
    [when when_value2 then statement_list2]...
    [else statement_list]
end case;

语法二

case 
when search_condition1 then statement_list1
[when search_condition2 then statement_list2]...
[else statement_list]
end case;

实例

create procedure p3(in month int)
begin
declare result varchar(10)
    case
        when month >=1 and month <=3 
        then result := '一季度';
    case 
          when month >=4 and month <=6 
        then result := '二季度';
    case
        when month >=7 and month <=9 
        then result := '三季度';
    case
        when month >=10 and month <=12 
        then result := '四季度';
    end case;

select result;
end;

while

while循环是有条件的循环控制语句

语法

while 条件 DO
    SQL逻辑
END while;

实例

create procedure p3(in number int)
begin
declare total int default 0;
while number>0 do
    set total := total+number
    set number := number-1
end while;
select number;
end;

repeat

有条件的循环控制语句，满足条件时候退出循环。

类似do while 会先执行一次。

语法

#先执行一次逻辑，如果满足退出循环，不满足退出循环
REPEAT
    SQL逻辑
    UNTIL 条件
END REPEAT;

实例

create procedure p3(in number int)
begin
declare total int default 0;
repeat
    set total := total+number
    set number := number-1
    UNTIL number <=0
end repeat;
select number;
end;

loop

实现简单循环，不在SQL逻辑中增加退出循环条件，可以用来实现简单的死循环。

Loop可以配合两个语句使用：

LEAVE：配合循环使用，退出循环 类似break

ITERATE：必须用在循环中，作用是跳过当前循环剩下的语句，直接进入下一次循环。类似continue

语法：

[begin_label:] LOOP
    SQL逻辑...
END LOOP [end_label];

实例1

create procedure p3(in number int)
begin

declare total int default 0;

sum:LOOP
    if(number<=0) then 
        LEAVE sum;
    end if;

set number := total+number;
set number := nunber-1

end loop sum;
select number;
end;

实例2

0-n的偶数相加结果

create procedure p3(in number int)
begin

declare total int default 0;

sum:LOOP
    if(number<=0) then 
        LEAVE sum;
    end if;
    
    if(number%2=1) then
        set number := nunber-1
        ITERATE sum;
    end if;

set number := total+number;
set number := nunber-1

end loop sum;
select number;
end;

游标cursor

游标CURSOR用来存储查询结果集的数据类型，

在存储过程和函数中可以使用游标对结果集进行循环的处理。游标的使用包括游标的声明、OPEN、FETCH和CLOSE。

语法

#声明游标
DECLARE 游标名称 CURSOR FOR 查询语句;

#打开游标
OPEN 游标名称;

#获取游标记录
FETCH 游标名称 INTO 变量[,变量];

#关闭游标
CLOSE 游标名称;

实例

查询出来的结果集放入到新表中。

create procedure p5(in uage int)
begin

#变量声明在游标之前
declare uname varchar(100);
declare upro varchar(100);

DECLARE result CURSOR FOR 
select name,profession from tb_user where age<=uage;

drop table if exists tb_user_pro;
create table if not exists tb_user_pro(
    id int primary key auto_increment,
    name varchar(100),
    profession varchar(100)
)

open result;
while true do 
    fetch r esult into uname,upro;
    insert ino tb_user_pro values(null,uname,upro);
end while;

close result;
end;

这里可以执行，但是会报错，因为while (true) 死循环，到最后一行插入数据后，继续获取下一行数据就会报错。这里使用条件处理程序处理错误

条件处理程序Handler

条件处理程序(Handler)可以用来定义在流程控制结构执行过程中遇到问题时相应的处理步骤。

语法：

DECLARE handler_action HANDLER FOR condition_value [,condition_value]... statement;

handler_action:
    CONTINUE:继续执行当前程序
    EXIT：终止执行当前程序
    
condition_value
    SQLSTATE sqlstate_value：状态码，如02000。
    状态码信息可以看Mysql的官方文档：https://dev.mysql.com/doc/mysql-errors/8.0/en/server-error-reference.html
    SQLWARNING：所有以01开头的SQLSTATE代码的简写
    NOT FOUND:所有以02开头的SQLSTATE代码的简写
    SQLEXCEPTION：所有没有被SQLWARNING或NOT FOUND捕获的SQLSTATE代码的简写

实例

这样就不会报错了

create procedure p5(in uage int)
begin

#变量声明在游标之前
declare uname varchar(100);
declare upro varchar(100);
#声明游标
DECLARE result CURSOR FOR 
select name,profession from tb_user where age<=uage;
#声明条件处理程序：直接退出并关闭游标
declare exit handler For NOT FOUND CLOSE result
#上面的语句和下面语句作用一样
#declare exit handler For sqlstate 02000 CLOSE result


#创建表
drop table if exists tb_user_pro;
create table if not exists tb_user_pro(
    id int primary key auto_increment,
    name varchar(100),
    profession varchar(100)
)

open result;
while true do 
    fetch r esult into uname,upro;
    insert ino tb_user_pro values(null,uname,upro);
end while;

close result;
end;

存储函数

介绍

存储函数是有返回值的存储过程，存储函数的参数只能是IN类型的。

缺点：必须得有返回值，建议使用存储过程

语法

CREATE FUNCTION 存储函数名称[(参数列表)]
RETURNS type [characteristic ...]
BEGIN
    SQL语句
    RETURN...;
END;

#这里的说明类型必须添加
characteristic说明
    DETERMINISTIC:相同的输入参数总是产生相同的结果
    NO SQL：不包含SQL语句
    READS SQL DATA：包含读取数据的语句，但不包含写入数据的语句。

实例

create function fun1(n int)

returns int DETERMINISTIC
    begin
        declare total int default 0;
        while n>0 do 
        set total := total + n
        set n := n - 1
     end while;
     return total;
end;

触发器

介绍

触发器是与表有关的数据库对象，只在insert/update/delete之前或之后，触发并执行触发器中定义的SQL语句集合。

触发器这种特性可以在数据库端确保数据的完整性，日志记录，数据效验等操作。

使用别名 OLD 和 NEW 来引用触发器中发生变化的记录内容，和其他数据库是相似的。现在触发器还指支持行级触发，不支持语句级触法。相应操作影响多少行，触发器就执行多少次。

INSERT 触发器  NEW表示将要或者已经新增的数据
UPDATE 触发器  OLD表示修改前的数据，NEW表示将要或已经修改的后的数据
DELETE 触发器  OLD表示将要或者已经删除的数据

语法

创建触发器

CREATE TRIGGER trigger_name
BEFORE|AFTER  INSERT|UPDATE|DELETE
ON table_name FOR EACH ROW  #行级触发器
BEGIN
    trigger_stmt;
END;

查看触发器

show TRIGGERS;

删除触发器

DROP TRIGGER [SCHEMA_NAME.] trigger_name
#如果不指定 SCHEMA_NAME，默认是当前数据库

实例

#创建日志表
create table user_logs(
id int(11) not null auto_increment,
    operation varchar(20) not null comment'操作时间，insert|delete|update',
    operate_time datetime not null comment'操作时间',
    operate_id int(11) not null comment '操作的ID',
    OPERATE_PARAMS VARCHAR(500) comment'操作参数',
    primary key(id)
)engine=innodb default charset=utf8;

insert触发器

create trigger tge()
after insert 
on tb_user for each row 
begin
    insert into user_logs
    values(null,'insert',now(),new.id,concat(new.id,',',new.name,',',.....))
end;

update触发器

create trigger tge()
after update 
on tb_user for each row 
begin
    insert into user_logs
    values(null,'update',now(),new.id,concat(old.id,',',old.name,',',new.id,',',new.name,',',....))
end;

delete触发器

create trigger tge()
after insert,update,delete 
on tb_user for each row 
begin
    insert into user_logs
    values(null,'delete',now(),old.id,concat(old.id,',',old.name,',',.....))
end;
